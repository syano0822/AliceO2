52  framework::AxisSpec multAxis = {tmpVecMult, "Multiplicity"};

54  framework::AxisSpec kTAxis = {kTBins, "#it{k}_{T} (GeV/#it{c})"};
55  framework::AxisSpec mTAxis = {mTBins, "#it{m}_{T} (GeV/#it{c}^{2})"};

57  mHistogramRegistry->add("relPairDist", ("; " + femtoObs + "; Entries").c_str(), o2::framework::kTH1F, {femtoObsAxis});
58  mHistogramRegistry->add("relPairkT", "; #it{k}_{T} (GeV/#it{c}); Entries", o2::framework::kTH1F, {kTAxis});
59  mHistogramRegistry->add("relPairkstarkT", ("; " + femtoObs + "; #it{k}_{T} (GeV/#it{c})").c_str(), o2::framework::kTH2F, {femtoObsAxis, kTAxis});
60  mHistogramRegistry->add("relPairkstarmT", ("; " + femtoObs + "; #it{m}_{T} (GeV/#it{c}^{2})").c_str(), o2::framework::kTH2F, {femtoObsAxis, mTAxis});
61  mHistogramRegistry->add("relPairkstarMult", ("; " + femtoObs + "; Multiplicity").c_str(), o2::framework::kTH2F, {femtoObsAxis, multAxis});
62  }

71  void setPair(T const& part1, T const& part2, const int mult)
72  {
73  float femtoObs;
74  if (mFemtoObs == femtoDreamContainer::Observable::kstar) {
75  femtoObs = FemtoDreamMath::getkstar(part1, mMassOne, part2, mMassTwo);
76  }
77  const float kT = FemtoDreamMath::getkT(part1, mMassOne, part2, mMassTwo);
78  const float mT = FemtoDreamMath::getmT(part1, mMassOne, part2, mMassTwo);
79 
80  if (mHistogramRegistry) {
81  mHistogramRegistry->fill(HIST("relPairDist"), femtoObs);
82  mHistogramRegistry->fill(HIST("relPairkT"), kT);
83  mHistogramRegistry->fill(HIST("relPairkstarkT"), femtoObs, kT);
84  mHistogramRegistry->fill(HIST("relPairkstarmT"), femtoObs, mT);
85  mHistogramRegistry->fill(HIST("relPairkstarMult"), femtoObs, mult);
86  }
87  }
