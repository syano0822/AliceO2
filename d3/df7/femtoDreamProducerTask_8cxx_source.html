59 template <typename T>
60 int getRowDaughters(int daughID, T const& vecID)
61 {
62  int rowInPrimaryTrackTableDaugh = -1;
63  for (size_t i = 0; i < vecID.size(); i++) {
64  if (vecID.at(i) == daughID) {
65  rowInPrimaryTrackTableDaugh = i;
66  break;
67  }
68  }
69  return rowInPrimaryTrackTableDaugh;
70 }

176  void process(aod::FilteredFullCollision const& col,
177  aod::FilteredFullTracks const& tracks,
178  o2::aod::V0Datas const& fullV0s)
179  {
180  if (!colCuts.isSelected(col)) {
181  return;
182  }
183  const auto vtxZ = col.posZ();
184  const auto mult = col.multV0M();
185  const auto spher = colCuts.computeSphericity(col, tracks);
186  colCuts.fillQA(col);
187  outputCollision(vtxZ, mult, spher);
188 
189  int childIDs[2] = {0, 0};
190  std::vector<int> tmpIDtrack;
191  float temptrack[2];
192  std::vector<float> temptrackPt;
193  std::vector<float> tempPostrackPt;
194 
195  for (auto& track : tracks) {
196  if (!trackCuts.isSelectedMinimal(track)) {
197  continue;
198  }
199  trackCuts.fillQA(track);
200  auto cutContainer = trackCuts.getCutContainer(track);
201  if (cutContainer > 0) {
202  trackCuts.fillCutQA(track, cutContainer);
203  outputTracks(outputCollision.lastIndex(), track.pt(), track.eta(), track.phi(), aod::femtodreamparticle::ParticleType::kTrack, cutContainer, track.dcaXY(), childIDs);
204  tmpIDtrack.push_back(track.globalIndex());
205  temptrackPt.push_back(track.pt());
206  temptrack[0] = outputTracks.lastIndex();
207  temptrack[1] = track.pt();
208  if (ConfDebugOutput) {
209  outputDebugTracks(outputCollision.lastIndex(),
210  track.sign(), track.tpcNClsFound(),
211  track.tpcNClsFindable(),
212  track.tpcNClsCrossedRows(), track.tpcNClsShared(), track.dcaXY(), track.dcaZ(),
213  track.tpcNSigmaEl(), track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(), track.tpcNSigmaDe(),
214  track.tofNSigmaEl(), track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(), track.tofNSigmaDe());
215  }
216  }
217  }
218 
219  for (auto& v0 : fullV0s) {
220  auto postrack = v0.posTrack_as<aod::FilteredFullTracks>();
221  auto negtrack = v0.negTrack_as<aod::FilteredFullTracks>();
222  if (!v0Cuts.isSelectedMinimal(col, v0, postrack, negtrack)) {
223  continue;
224  }
225  v0Cuts.fillQA(col, v0);
226  auto cutContainerV0 = v0Cuts.getCutContainer(col, v0, postrack, negtrack);
227  if ((cutContainerV0.at(0) > 0) && (cutContainerV0.at(1) > 0) && (cutContainerV0.at(2) > 0)) {
228  int postrackID = v0.posTrackId();
229  int rowInPrimaryTrackTablePos = -1;
230  rowInPrimaryTrackTablePos = getRowDaughters(postrackID, tmpIDtrack);
231  childIDs[0] = rowInPrimaryTrackTablePos;
232  childIDs[1] = 0;
233  ROOT::Math::PxPyPzMVector postrackVec(v0.pxpos(), v0.pypos(), v0.pzpos(), 0.);
234  ROOT::Math::PxPyPzMVector negtrackVec(v0.pxneg(), v0.pyneg(), v0.pzneg(), 0.);
235  outputTracks(outputCollision.lastIndex(), postrackVec.Pt(), postrackVec.Eta(), postrackVec.Phi(), aod::femtodreamparticle::ParticleType::kV0Child, cutContainerV0.at(1), 0., childIDs);
236  const int rowOfPosTrack = outputTracks.lastIndex();
237  int negtrackID = v0.negTrackId();
238  int rowInPrimaryTrackTableNeg = -1;
239  rowInPrimaryTrackTableNeg = getRowDaughters(negtrackID, tmpIDtrack);
240  childIDs[0] = 0;
241  childIDs[1] = rowInPrimaryTrackTableNeg;
242  outputTracks(outputCollision.lastIndex(), negtrackVec.Pt(), negtrackVec.Eta(), negtrackVec.Phi(), aod::femtodreamparticle::ParticleType::kV0Child, cutContainerV0.at(2), 0., childIDs);
243  const int rowOfNegTrack = outputTracks.lastIndex();
244  int indexChildID[2] = {rowOfPosTrack, rowOfNegTrack};
245  outputTracks(outputCollision.lastIndex(), v0.pt(), v0.eta(), v0.phi(), aod::femtodreamparticle::ParticleType::kV0, cutContainerV0.at(0), v0.v0cosPA(col.posX(), col.posY(), col.posZ()), indexChildID);
246  }
247  }
248  }
