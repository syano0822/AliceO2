107 template <typename C, typename V, typename T>
108 bool FemtoDreamV0Selection::isSelectedMinimal(C const& col, V const& v0, T const& posTrack, T const& negTrack)
109 {
110  const auto signPos = posTrack.sign();
111  const auto signNeg = negTrack.sign();
112  // printf("pos sign = %i --- neg sign = %i\n", signPos, signNeg);
113  if (signPos < 0 || signNeg > 0) {
114  printf("-Something wrong in isSelectedMinimal--\n");
115  printf("ERROR - Wrong sign for V0 daughters\n");
116  }
117  const float pT = v0.pt();
118  const std::vector<float> decVtx = {v0.x(), v0.y(), v0.z()};
119  const float tranRad = v0.v0radius();
120  const float dcaDaughv0 = v0.dcaV0daughters();
121  const float cpav0 = v0.v0cosPA(col.posX(), col.posY(), col.posZ());
122 
124  const static int nPtV0MinSel = getNSelections(femtoDreamV0Selection::kpTV0Min);
125  const static int nPtV0MaxSel = getNSelections(femtoDreamV0Selection::kpTV0Max);
126  const static int nDCAV0DaughMax = getNSelections(femtoDreamV0Selection::kDCAV0DaughMax);
127  const static int nCPAV0Min = getNSelections(femtoDreamV0Selection::kCPAV0Min);
128  const static int nTranRadV0Min = getNSelections(femtoDreamV0Selection::kTranRadV0Min);
129  const static int nTranRadV0Max = getNSelections(femtoDreamV0Selection::kTranRadV0Max);
130  const static int nDecVtxMax = getNSelections(femtoDreamV0Selection::kDecVtxMax);
131 
132  const static float pTV0Min = getMinimalSelection(femtoDreamV0Selection::kpTV0Min, femtoDreamSelection::kLowerLimit);
133  const static float pTV0Max = getMinimalSelection(femtoDreamV0Selection::kpTV0Max, femtoDreamSelection::kUpperLimit);
134  const static float DCAV0DaughMax = getMinimalSelection(femtoDreamV0Selection::kDCAV0DaughMax, femtoDreamSelection::kUpperLimit);
135  const static float CPAV0Min = getMinimalSelection(femtoDreamV0Selection::kCPAV0Min, femtoDreamSelection::kLowerLimit);
136  const static float TranRadV0Min = getMinimalSelection(femtoDreamV0Selection::kTranRadV0Min, femtoDreamSelection::kLowerLimit);
137  const static float TranRadV0Max = getMinimalSelection(femtoDreamV0Selection::kTranRadV0Max, femtoDreamSelection::kUpperLimit);
138  const static float DecVtxMax = getMinimalSelection(femtoDreamV0Selection::kDecVtxMax, femtoDreamSelection::kAbsUpperLimit);
139 
140  if (nPtV0MinSel > 0 && pT < pTV0Min) {
141  return false;
142  }
143  if (nPtV0MaxSel > 0 && pT > pTV0Max) {
144  return false;
145  }
146  if (nDCAV0DaughMax > 0 && dcaDaughv0 > DCAV0DaughMax) {
147  return false;
148  }
149  if (nCPAV0Min > 0 && cpav0 < CPAV0Min) {
150  return false;
151  }
152  if (nTranRadV0Min > 0 && tranRad < TranRadV0Min) {
153  return false;
154  }
155  if (nTranRadV0Max > 0 && tranRad > TranRadV0Max) {
156  return false;
157  }
158  for (int i = 0; i < decVtx.size(); i++) {
159  if (nDecVtxMax > 0 && decVtx.at(i) > DecVtxMax) {
160  return false;
161  }
162  }
163  if (!PosDaughTrack.isSelectedMinimal(posTrack)) {
164  return false;
165  }
166  if (!NegDaughTrack.isSelectedMinimal(negTrack)) {
167  return false;
168  }
169  return true;
170 }

173 template <typename C, typename V, typename T>
174 std::vector<uint64_t> FemtoDreamV0Selection::getCutContainer(C const& col, V const& v0, T const& posTrack, T const& negTrack)
175 {
176  uint64_t outputPosTrack = PosDaughTrack.getCutContainer(posTrack);
177  uint64_t outputNegTrack = NegDaughTrack.getCutContainer(negTrack);
178  uint64_t output = 0;
179  size_t counter = 0;
180 
181  const auto pT = v0.pt();
182  const auto tranRad = v0.v0radius();
183  const auto dcaDaughv0 = v0.dcaV0daughters();
184  const auto cpav0 = v0.v0cosPA(col.posX(), col.posY(), col.posZ());
185  const std::vector<float> decVtx = {v0.x(), v0.y(), v0.z()};
186 
187  float observable;
188  for (auto& sel : mSelections) {
189  const auto selVariable = sel.getSelectionVariable();
190  if (selVariable == femtoDreamV0Selection::kDecVtxMax) {
191  for (size_t i = 0; i < decVtx.size(); ++i) {
192  auto decVtxValue = decVtx.at(i);
193  sel.checkSelectionSetBit(decVtxValue, output, counter);
194  }
195  } else {
196  switch (selVariable) {
197  case (femtoDreamV0Selection::kpTV0Min):
198  case (femtoDreamV0Selection::kpTV0Max):
199  observable = pT;
200  break;
201  case (femtoDreamV0Selection::kDCAV0DaughMax):
202  observable = dcaDaughv0;
203  break;
204  case (femtoDreamV0Selection::kCPAV0Min):
205  observable = cpav0;
206  break;
207  case (femtoDreamV0Selection::kTranRadV0Min):
208  case (femtoDreamV0Selection::kTranRadV0Max):
209  observable = tranRad;
210  break;
211  case (femtoDreamV0Selection::kDecVtxMax):
212  break;
213  }
214  sel.checkSelectionSetBit(observable, output, counter);
215  }
216  }
217  return {{outputPosTrack, outputNegTrack, output}};
218 }
