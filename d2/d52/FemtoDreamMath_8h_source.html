37  template <typename T>
38  static float getkstar(const T& part1, const float mass1, const T& part2, const float mass2)
39  {
40  ROOT::Math::PtEtaPhiMVector vecpart1(part1.pt(), part1.eta(), part1.phi(), mass1);
41  ROOT::Math::PtEtaPhiMVector vecpart2(part2.pt(), part2.eta(), part2.phi(), mass2);
42 
43  ROOT::Math::PtEtaPhiMVector trackSum = vecpart1 + vecpart2;
44 
45  float beta = trackSum.Beta();
46  float betax = beta * std::cos(trackSum.Phi()) * std::sin(trackSum.Theta());
47  float betay = beta * std::sin(trackSum.Phi()) * std::sin(trackSum.Theta());
48  float betaz = beta * std::cos(trackSum.Theta());
49 
50  ROOT::Math::PxPyPzMVector PartOneCMS(vecpart1);
51  ROOT::Math::PxPyPzMVector PartTwoCMS(vecpart2);
52 
53  ROOT::Math::Boost boostPRF = ROOT::Math::Boost(-betax, -betay, -betaz);
54  PartOneCMS = boostPRF(PartOneCMS);
55  PartTwoCMS = boostPRF(PartTwoCMS);
56 
57  ROOT::Math::PxPyPzMVector trackRelK = PartOneCMS - PartTwoCMS;
58  return 0.5 * trackRelK.P();
59  }
